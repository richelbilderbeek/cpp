  assert(s[0] == '<');   assert(s[s.size() - 1] == '>');   assert(s.find('>') != std::string::npos);   const int tag_name_sz = static_cast<int>(s.find('>')) - 1;   const std::string tag_name = s.substr(1,tag_name_sz); ///   } /// }; /// ///The data can be converted back with PtrsToXml template <class T> const std::pair<     std::string,     std::vector<boost::shared_ptr<T>>   >   XmlToPtrs(     const std::string& s,     const std::function<const boost::shared_ptr<T>(const std::string&)> str_to_ptr_function   ) {   assert(!s.empty());
